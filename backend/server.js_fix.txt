        let extractedText = '';
        try {
            const dataBuffer = fs.readFileSync(req.file.path);
            const pdfData = new Uint8Array(dataBuffer);
            const pdfjsLib = await import('pdfjs-dist');
            const loadingTask = pdfjsLib.getDocument({ data: pdfData, useWorkerFetch: false, isEvalSupported: false });
            const pdf = await loadingTask.promise;
            let fullText = '';
            for (let i = 1; i <= pdf.numPages; i++) {
                const page = await pdf.getPage(i);
                const textContent = await page.getTextContent();
                fullText += textContent.items.map(item => item.str).join(' ') + '\n';
            }
            extractedText = fullText.trim();
            if (!extractedText) {
                extractedText = `### ⚠️ NO TEXT DETECTED\n- The file **${req.file.originalname}** was uploaded, but no text could be extracted.\n- **Possible Reason:** This PDF might be a scanned image (picture of text).\n- **Solution:** For auto-generated key points, please use a PDF that contains actual text, or upload images directly for OCR processing.`;
            }
        } catch (pErr) { 
            console.error("Extraction error:", pErr);
            extractedText = `### ⚠️ EXTRACTION ERROR\n- Failed to extract text from ${req.file.originalname}. The file was uploaded successfully.`;
        }

        if (extractedText && extractedText.trim().length > 0 && !extractedText.startsWith('###')) {
            const sentences = extractedText.split(/[.!?]\s+/).filter(s => s.trim().length > 30);
            const count = sentences.length;
            const sampled = [
              ...sentences.slice(0, 2),
              ...sentences.slice(Math.floor(count/2), Math.floor(count/2) + 2),
              ...sentences.slice(-2)
            ];
            const narrativeSummary = sampled.map(s => s.trim()).join('. ') + '.';
            const formattedContent = `### 📖 FULL TOPIC EXPLANATION\n\n${narrativeSummary}\n\n---`;
            db.prepare('UPDATE notes SET content = ?, updatedAt = CURRENT_TIMESTAMP WHERE id = ?').run(formattedContent, id);
        } else {
            db.prepare('UPDATE notes SET content = ?, updatedAt = CURRENT_TIMESTAMP WHERE id = ?').run(extractedText, id);
